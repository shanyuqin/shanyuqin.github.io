iOS开发笔记——WKWebViewAPI整理

WKWebView的属性和方法

WKWebView头文件如下：

// webView的配置
@property (nonatomic, readonly, copy) WKWebViewConfiguration *configuration;

// 导航代理
@property (nullable, nonatomic, weak) id <WKNavigationDelegate> navigationDelegate;

// UI交互代理
@property (nullable, nonatomic, weak) id <WKUIDelegate> UIDelegate;

// 页面前进和后退列表
@property (nonatomic, readonly, strong) WKBackForwardList *backForwardList;

//默认初始化方法
- (instancetype)initWithFrame:(CGRect)frame configuration:(WKWebViewConfiguration *)configuration NS_DESIGNATED_INITIALIZER;

//不会用到
- (nullable instancetype)initWithCoder:(NSCoder *)coder NS_DESIGNATED_INITIALIZER;

//加载请求
- (nullable WKNavigation *)loadRequest:(NSURLRequest *)request;

//加载URL
- (nullable WKNavigation *)loadFileURL:(NSURL *)URL allowingReadAccessToURL:(NSURL *)readAccessURL API_AVAILABLE(macosx(10.11), ios(9.0));

//加载Html
- (nullable WKNavigation *)loadHTMLString:(NSString *)string baseURL:(nullable NSURL *)baseURL;

//加载Data
- (nullable WKNavigation *)loadData:(NSData *)data MIMEType:(NSString *)MIMEType characterEncodingName:(NSString *)characterEncodingName baseURL:(NSURL *)baseURL API_AVAILABLE(macosx(10.11), ios(9.0));

//前进或者后退到指定页面
- (nullable WKNavigation *)goToBackForwardListItem:(WKBackForwardListItem *)item;

//页面标题，只读属性，修改需通过KVO
@property (nullable, nonatomic, readonly, copy) NSString *title;

//当前请求的URL，只读属性，修改需通过KVO
@property (nullable, nonatomic, readonly, copy) NSURL *URL;

//是否正在加载，只读属性，修改需通过KVO
@property (nonatomic, readonly, getter=isLoading) BOOL loading;

//加载进度 【0，1】
@property (nonatomic, readonly) double estimatedProgress;

//标示页面中的所有资源是否通过安全加密链接来加载（这里说的是https么？）
@property (nonatomic, readonly) BOOL hasOnlySecureContent;

//当前导航的证书链（什么是证书链）
@property (nonatomic, readonly, nullable) SecTrustRef serverTrust API_AVAILABLE(macosx(10.12), ios(10.0));

//能否执行goBack操作
@property (nonatomic, readonly) BOOL canGoBack;

//能否执行goForward操作
@property (nonatomic, readonly) BOOL canGoForward;

//返回上以页面，如果不能返回则什么都不做
- (nullable WKNavigation *)goBack;

//进入下一页面，如果不能进入则什么都不做
- (nullable WKNavigation *)goForward;

//重载当前页面
- (nullable WKNavigation *)reload;

//从原始URL重载
- (nullable WKNavigation *)reloadFromOrigin;

//停止加载数据
- (void)stopLoading;

//执行JS源码中的某个方法
- (void)evaluateJavaScript:(NSString *)javaScriptString completionHandler:(void (^ _Nullable)(_Nullable id, NSError * _Nullable error))completionHandler;

//是否支持左、右滑动手势来前进下一页面或后退到上一页面
@property (nonatomic) BOOL allowsBackForwardNavigationGestures;

//自定义 userAgent，如果没有则为nil （userAgent是什么鬼）
@property (nullable, nonatomic, copy) NSString *customUserAgent API_AVAILABLE(macosx(10.11), ios(9.0));

//是否允许链接预览 iOS默认为NO
@property (nonatomic) BOOL allowsLinkPreview API_AVAILABLE(macosx(10.11), ios(9.0));

#if TARGET_OS_IPHONE
//
@property (nonatomic, readonly, strong) UIScrollView *scrollView;
#endif

#if !TARGET_OS_IPHONE
//是否支持放大手势，默认为NO
@property (nonatomic) BOOL allowsMagnification;

//放大因子，放大倍数 默认为1
@property (nonatomic) CGFloat magnification;

//根据设置的书房因子来缩放页面，并居中显示结果在制定的点
- (void)setMagnification:(CGFloat)magnification centeredAtPoint:(CGPoint)point;
#endif

WKWebViewConfiguration - webView的配置

WKWebViewConfiguration *config = [[WKWebViewConfiguration alloc] init];

configuration中的一些属性介绍：

WKPreferences偏好设置

// 设置偏好设置
config.preferences = [[WKPreferences alloc] init];
// 最小字体大小 默认为0
config.preferences.minimumFontSize = 10;
// 是否可以使用javaScri默认认为YES
config.preferences.javaScriptEnabled = YES;
// 在iOS上默认为NO，表示不能自动通过窗口打开
config.preferences.javaScriptCanOpenWindowsAutomatically = NO;

WKProcessPool内容处理池

WKProcessPool没有公开的属性或者方法，用处暂时不清楚
config.processPool = [[WKProcessPool alloc] init];
实际上我们也不是必须要创建这个处理池来完成webView的配置

WKUserContentController内容交互控制器

这个类用于实现JS和webView的内容交互

//只读属性，所有添加的WKUserScript都在这里可以获取到
@property (nonatomic, readonly, copy) NSArray<WKUserScript *> *userScripts;
//注入JS脚本
- (void)addUserScript:(WKUserScript *)userScript;
//移除所有注入的JS
- (void)removeAllUserScripts;
//添加scriptMessageHandler，之后可以通过
//window.webkit.messageHandlers.<name.postmessage(<messageBody>)>发送消息
//比如JS调用我们原生的方法，就可以通过这种方式
- (void)addScriptMessageHandler:(id <WKScriptMessageHandler>)scriptMessageHandler name:(NSString *)name;
//根据name移除注入的scriptMessageHandler
- (void)removeScriptMessageHandlerForName:(NSString *)name;
WKUserScript

WKUserContentController注入的JS就是一个WKUserScript对象，属性和方法如下： 

//JS源码
@property (nonatomic, readonly, copy) NSString *source;
//JS注入时间
@property (nonatomic, readonly) WKUserScriptInjectionTime injectionTime;
//表示JS是否应该注入到所有的frames种还是只有main frame
@property (nonatomic, readonly, getter=isForMainFrameOnly) BOOL forMainFrameOnly;
//初始化方法
- (instancetype)initWithSource:(NSString *)source injectionTime:(WKUserScriptInjectionTime)injectionTime forMainFrameOnly:(BOOL)forMainFrameOnly;

WKUserScriptInjectionTime

typedef NS_ENUM(NSInteger, WKUserScriptInjectionTime) {
    WKUserScriptInjectionTimeAtDocumentStart,//文档开始加载时注入
    WKUserScriptInjectionTimeAtDocumentEnd //加载结束时注入
} NS_ENUM_AVAILABLE(10_10, 8_0);

这里有两个概念没有提到分别是WKWebsiteDataStore（存储的web内容）和WKWebsiteDataRecord（website的数据存储记录类型），他们都是iOS9.0以后才可用，之后会对这里进行补充，暂时略过

WKNavigationDelegate

@protocol WKNavigationDelegate <NSObject>

@optional

// 决定导航的动作，通常用于处理跨域的链接能否导航。WebKit对跨域进行了安全检查限制，不允许跨域，因此我们要对不能跨域的链接
// 单独处理。但是，对于Safari是允许跨域的，不用这么处理。
// 这个是决定是否Request
- (void)webView:(WKWebView *)webView decidePolicyForNavigationAction:(WKNavigationAction *)navigationAction decisionHandler:(void (^)(WKNavigationActionPolicy))decisionHandler;

// 决定是否接收响应
// 这个是决定是否接收response
// 要获取response，通过WKNavigationResponse对象获取
- (void)webView:(WKWebView *)webView decidePolicyForNavigationResponse:(WKNavigationResponse *)navigationResponse decisionHandler:(void (^)(WKNavigationResponsePolicy))decisionHandler;

// 当main frame的导航开始请求时，会调用此方法
- (void)webView:(WKWebView *)webView didStartProvisionalNavigation:(null_unspecified WKNavigation *)navigation;

// 当main frame接收到服务重定向时，会回调此方法
- (void)webView:(WKWebView *)webView didReceiveServerRedirectForProvisionalNavigation:(null_unspecified WKNavigation *)navigation;

// 当main frame开始加载数据失败时，会回调
- (void)webView:(WKWebView *)webView didFailProvisionalNavigation:(null_unspecified WKNavigation *)navigation withError:(NSError *)error;

// 当main frame的web内容开始到达时，会回调
- (void)webView:(WKWebView *)webView didCommitNavigation:(null_unspecified WKNavigation *)navigation;

// 当main frame导航完成时，会回调
- (void)webView:(WKWebView *)webView didFinishNavigation:(null_unspecified WKNavigation *)navigation;

// 当main frame最后下载数据失败时，会回调
- (void)webView:(WKWebView *)webView didFailNavigation:(null_unspecified WKNavigation *)navigation withError:(NSError *)error;

// 这与用于授权验证的API，与AFN、UIWebView的授权验证API是一样的
- (void)webView:(WKWebView *)webView didReceiveAuthenticationChallenge:(NSURLAuthenticationChallenge *)challenge completionHandler:(void (^)(NSURLSessionAuthChallengeDisposition disposition, NSURLCredential *__nullable credential))completionHandler;

// 当web content处理完成时，会回调
- (void)webViewWebContentProcessDidTerminate:(WKWebView *)webView NS_AVAILABLE(10_11, 9_0);

@end
